**(a) Create `src/lib/repoSize.ts`:**

```typescript
// ARCH-01 observability helper — reports the size of the .git directory via
// `du -sh .git` shell-out. Single-responsibility module so writeStepSummary
// stays pure-ish and easy to unit-test with a Recorder fixture.
//
// Invariants:
//   1. NEVER-THROWS — a `du` failure (missing binary, non-git cwd, permission
//      denied) returns 0 and logs a console.warn. This preserves the
//      end-to-end never-throw contract of writeStepSummary (Phase 3 Pitfall 10):
//      observability must not fail the workflow.
//   2. SYNC-ISH — one shell-out per run (at the end, in the finally block of
//      runPipeline). No caching, no memoization. Size can change between runs.
//   3. CWD = process.cwd() — the helper trusts the caller's cwd. GHA sets
//      cwd to the repo root on actions/checkout@v5, so `.git` is discoverable.
//
// Output units: bytes. Byte-to-MB conversion is the CALLER'S responsibility
// (keeps this module unit-test-friendly — we assert exact byte counts).

import { execFile } from 'node:child_process';
import { promisify } from 'node:util';
import { scrubSecrets } from '../util/logging.js';

const execFileP = promisify(execFile);

export async function getGitDirBytes(): Promise<number> {
  try {
    // `du -sk .git` outputs size in 1024-byte blocks, portable across GNU/BSD du.
    // Parse the first whitespace-separated field as an integer.
    const { stdout } = await execFileP('du', ['-sk', '.git'], { cwd: process.cwd() });
    const kb = Number.parseInt(stdout.trim().split(/\s+/)[0] ?? '0', 10);
    if (!Number.isFinite(kb) || kb < 0) return 0;
    return kb * 1024;
  } catch (err) {
    // ARCH-01 observability failure must not break the workflow (Phase 3
    // Pitfall 10 inheritance). Report zero and log a scrubbed warning.
    console.warn(
      `[repo-size] du -sk .git failed: ${scrubSecrets((err as Error).message)}`,
    );
    return 0;
  }
}
```

import type { GlobalCounters } from './recorder.js';

// Add above the writeStepSummary export, below the existing imports:

// Phase 5 D-04 counter row rendering — three passive trigger counters with
// threshold-flag prefix (⚠) when values cross 80% of their budget.
// Thresholds:
//   QUOTA-01: geminiItems ≥80 of 100/day Gemini Flash free-tier RPD.
//   ARCH-01:  gitDirBytes ≥400 MB of 500 MB budget.
//   CACHE-01: summaryRetries ≥5 attempts per run.
// Flag character is locked as ⚠ across all three rows for operator
// scannability (05-CONTEXT.md specifics).
const QUOTA_FLAG_THRESHOLD = 80;
const ARCH_FLAG_THRESHOLD_BYTES = 400_000_000;  // 400 MB (decimal, matches `du` reporting convention)
const CACHE_FLAG_THRESHOLD = 5;

function renderCounterRows(counters: GlobalCounters): string {
  const geminiFlag = counters.geminiItems >= QUOTA_FLAG_THRESHOLD ? '⚠ ' : '';
  const archFlag = counters.gitDirBytes >= ARCH_FLAG_THRESHOLD_BYTES ? '⚠ ' : '';
  const cacheFlag = counters.summaryRetries >= CACHE_FLAG_THRESHOLD ? '⚠ ' : '';
  const gitMb = Math.floor(counters.gitDirBytes / 1_000_000);
  return [
    `${geminiFlag}Gemini items today: ${counters.geminiItems}/100`,
    `${archFlag}Repo .git size: ${gitMb} MB / 500 MB budget`,
    `${cacheFlag}Summary retries this run: ${counters.summaryRetries}`,
  ].join('\n');
}

export async function writeStepSummary(
  recorder: Recorder,
  firms: FirmConfig[],
): Promise<void> {
  const path = process.env.GITHUB_STEP_SUMMARY;
  if (!path) return;

    );
  }
}
