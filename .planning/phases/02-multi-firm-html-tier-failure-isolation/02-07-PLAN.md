<objective>
Extend the D-09 bootstrap guard in `dedupAll` so that a firm whose prior state is `{urls: [], lastNewAt: null}` is treated the same as a firm that is entirely absent from state. Currently only `!priorFirm` (undefined) triggers the bootstrap branch; a manually-edited empty entry OR a firm that never produced a new item (e.g., Skadden monthly) between being added and its state-writer run would bypass the guard and flood the digest with the whole back-catalog on the NEXT run.

Purpose: DEDUP-05 success criterion #3 in ROADMAP — "Adding a brand-new firm to `config/firms.yaml` and running once bootstraps its seen-URL set (no back-catalog flood in the next digest)." Phase 1 handles the `undefined` path; Phase 2 must also cover the Pitfall 6 edge case where `priorFirm` is defined but structurally empty.

Output: one-line logical extension to `dedup.ts` (OR-ing an additional empty-state check onto the existing `!priorFirm` guard), one new unit test in `dedup.test.ts`, zero regressions on the existing 6 tests. The change is deliberately minimal — we're hardening the guard, not rearchitecting dedup.

```typescript
// TDD coverage for src/state/writer.ts — Phase 2 D-P2-08 empty-state
// bootstrap mirror.
//
// The writer's bootstrap branch MUST treat an empty-state firm
// ({urls:[], lastNewAt:null}) the same as a missing firm — seeding
// urls from r.raw. Without this, dedup bootstraps (new:[]) but writer
// leaves urls:[] in place, and the empty-state repeats forever.

import { describe, it, expect, beforeEach, afterEach, vi } from 'vitest';
import { mkdtemp, readFile, rm, writeFile } from 'node:fs/promises';
import { tmpdir } from 'node:os';
import { join } from 'node:path';
import { writeState } from '../../src/state/writer.js';
import type { FirmConfig, FirmResult, SeenState } from '../../src/types.js';

const cooley: FirmConfig = {
  id: 'cooley',
  name: 'Cooley',
  language: 'en',
  type: 'rss',
  url: 'https://cooley.com/feed',
  timezone: 'America/Los_Angeles',
  enabled: true,
  timeout_ms: 20000,
};

function makeResult(raw: Array<{ url: string; title: string }>): FirmResult {
  return {
    firm: cooley,
    raw: raw.map((r) => ({
      firmId: 'cooley',
      title: r.title,
      url: r.url,
      language: 'en',
    })),
    new: [],
    summarized: [],
    durationMs: 0,
  };
}

describe('writeState (D-P2-08 empty-state bootstrap mirror)', () => {
  let dir: string;
  let statePath: string;
  const origDryRun = process.env.DRY_RUN;

  beforeEach(async () => {
    dir = await mkdtemp(join(tmpdir(), 'ln-writer-'));
    statePath = join(dir, 'seen.json');
    delete process.env.DRY_RUN;
  });

  afterEach(async () => {
    if (origDryRun === undefined) delete process.env.DRY_RUN;
    else process.env.DRY_RUN = origDryRun;
    await rm(dir, { recursive: true, force: true });
  });

  it('empty-state firm ({urls:[], lastNewAt:null}) seeds urls from r.raw (B1 mirror)', async () => {
    const prior: SeenState = {
      version: 1,
      lastUpdated: '2026-04-10T00:00:00.000Z',
      firms: { cooley: { urls: [], lastNewAt: null } },
    };
    const results = [
      makeResult([
        { url: 'https://cooley.com/a', title: 'A' },
        { url: 'https://cooley.com/b', title: 'B' },
      ]),
    ];
    await writeState(prior, results, statePath);
    const next: SeenState = JSON.parse(await readFile(statePath, 'utf8'));
    expect(next.firms.cooley.urls).toEqual([
      'https://cooley.com/a',
      'https://cooley.com/b',
    ]);
    // lastNewAt advances from null to SOMETHING (either a publishedAt or now())
    expect(next.firms.cooley.lastNewAt).not.toBeNull();
  });

  it('non-empty prior urls — writer does NOT reseed, preserves prior', async () => {
    const prior: SeenState = {
      version: 1,
      lastUpdated: '2026-04-10T00:00:00.000Z',
      firms: { cooley: { urls: ['https://cooley.com/old'], lastNewAt: null } },
    };
    const results = [makeResult([])];  // no new items this run
    // r.summarized is [] too so no merge additions
    await writeState(prior, results, statePath);
    const next: SeenState = JSON.parse(await readFile(statePath, 'utf8'));
    expect(next.firms.cooley.urls).toEqual(['https://cooley.com/old']);
  });
});
```
