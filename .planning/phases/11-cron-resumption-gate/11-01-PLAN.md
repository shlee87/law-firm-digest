    **Step 1: Add restoreFetchHost to src/scrapers/util.ts**

    Add after the canonicalizeUrl export (around line 120), as an exported function:

    ```typescript
    /**
     * Restore the `www.` prefix on a fetch-round-trip URL when the firm's
     * canonical URL has `www.` but canonicalizeUrl stripped it from the item URL.
     *
     * This resolves the bkl / kim-chang pattern (Phase 7-05/07-06): both firms'
     * TLS certs cover only `www.<domain>`, so bare-apex URLs produced by
     * canonicalizeUrl fail with ERR_CERT_COMMON_NAME_INVALID (kim-chang) or
     * redirect to homepage (bkl). The DEDUP-stored canonical URL is NOT changed —
     * only the URL used in the fetch round-trip is restored.
     *
     * @param itemUrl  Canonical item URL (www. may have been stripped).
     * @param firmUrl  The firm.url field from config (un-canonicalized, may carry www.).
     * @returns        itemUrl with www. restored iff firm.url carries it and the
     *                 item hostname is the www.-stripped form of firm.hostname.
     *                 Returns itemUrl unchanged in all other cases.
     */
    export function restoreFetchHost(itemUrl: string, firmUrl: string): string {
      const item = new URL(itemUrl);
      const firm = new URL(firmUrl);
      if (
        firm.hostname.startsWith('www.') &&
        firm.hostname.slice(4) === item.hostname
      ) {
        item.hostname = firm.hostname;
        return item.toString();
      }
      return itemUrl;
    }
    ```
